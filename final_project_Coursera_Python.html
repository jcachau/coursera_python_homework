The Assignment¶

Take a ZIP file) of images and process them, using a library built into python that you need to learn how to use. A ZIP file takes several different files and compresses them, thus saving space, into one single file. The files in the ZIP file we provide are newspaper images (like you saw in week 3). Your task is to write python code which allows one to search through the images looking for the occurrences of keywords and faces. E.g. if you search for "pizza" it will return a contact sheet of all of the faces which were located on the newspaper page which mentions "pizza". This will test your ability to learn a new (library), your ability to use OpenCV to detect faces, your ability to use tesseract to do optical character recognition, and your ability to use PIL to composite images together into contact sheets.

Each page of the newspapers is saved as a single PNG image in a file called images.zip. These newspapers are in english, and contain a variety of stories, advertisements and images. Note: This file is fairly large (~200 MB) and may take some time to work with, I would encourage you to use small_img.zip for testing.

fnt = ImageFont.truetype("readonly/fanwood-webfont.ttf",size=15)

# function definition to treat images

# converts from  CV coordinates to PIL coordinates
def convert_coord(cv_XY):
    return (cv_XY[0],cv_XY[1],cv_XY[0]+cv_XY[2],cv_XY[1]+cv_XY[3])

# extracts recognized faces as cropped images and puts them in a list (to be used to build the contact sheet)
def extract_faces(faces_to_show,image_to_extract):
    image_faces_list = []
    for item in faces_to_show:
        coord_draw = convert_coord(item)
        image_cropped = image_to_extract.crop(coord_draw)
        image_faces_list.append(image_cropped)
    return image_faces_list    

# makes individual contact sheets for each file. Inserts specific text if no face found

def make_contact_sheets(name_of_file,list_of_picture,width=500,height_row=100,height_text=20,Npic_byrow=5):

    N = len(list_of_picture)
    N_rows = ((N-1)//Npic_byrow+1)
 
    Height_sheet = N_rows*height_row
    Width_sheet = width
    Height_unit = height_row
    Width_unit = int(width/Npic_byrow)
    if list_of_picture==[]:
        output_sheet=Image.new("RGB", (Width_sheet+20, 2*height_text),color="white")
        draw = ImageDraw.Draw(output_sheet)
        draw.text((5,0),
                  "Results found in file {} \nBut there were no faces in that file!".format(name_of_file),
                  font=fnt,fill=(0,0,0))
        
    else:    
        first_image=list_of_picture[0]  

        output_sheet_base=Image.new(first_image.mode,
                               (Width_sheet, Height_sheet))
        
        x,y=0,0
        
        for item in list_of_picture:
            if item.width >=Width_unit or item.height>Height_unit:
                pic_to_add = item.resize((Width_unit,Height_unit))
            else:
                pic_to_add = item
            output_sheet_base.paste(pic_to_add, (x, y))
            if x+Width_unit == Width_sheet:
                     x=0
                     y=y+Height_unit
            else:
                     x=x+Width_unit
      # display(output_sheet_base)         
        output_sheet=Image.new("RGB",
                               (output_sheet_base.width+20,output_sheet_base.height+height_text+5),
                                color="white")
        draw = ImageDraw.Draw(output_sheet)
        draw.text((5,0),"Results found in file {}".format(name_of_file),
                  font=fnt,fill=(0,0,0))
        output_sheet.paste(output_sheet_base,(5,height_text))
    return output_sheet

# main function: takes a dictionary of image, identifies faces, calls subfunction to issue individual contact sheets (by file)
# and then put all of these together to produce a master contact sheet for all files relevant to the given name

def produce_contact_sheet(img_dict):
    
        contact_sheets=[]
            
        for name in img_dict.keys():
            #print(name)
            name_pic = name
            img = img_dict[name]
            img = img.convert("RGB")
            img_gray = img.convert("L")
            img_gray.save('greyscale.jpg')
            cv_image = cv.imread('greyscale.jpg')
            img_byte = cv.cvtColor(cv_image, cv.COLOR_BGR2GRAY)
            
            Scale_Factor = 1.3
            Min_neighbours=6
            faces = face_cascade.detectMultiScale(img_byte,scaleFactor=Scale_Factor, minNeighbors=Min_neighbours)
        
            if len(faces) ==0:
                list_faces=[]
            else:
                faces_list = faces.tolist()
               # print(len(faces_list))
                list_faces = extract_faces(faces_list,img)
            output_sheet=make_contact_sheets(name,list_faces)
            contact_sheets.append(output_sheet)
            
        Height_complete_sheet = sum([x.height for x in contact_sheets])
        Width_complete_sheet = max([x.width for x in contact_sheets])
        complete_sheet = Image.new("RGB",(Width_complete_sheet,Height_complete_sheet),color="white")
        x,y=0,0
        for item in contact_sheets:
            complete_sheet.paste(item,(x,y))
            y= y+item.height
        
        return complete_sheet
